
This what your config file should look like.  A sample is also provided.

<?xml version="1.0" encoding="UTF-8"?>
<config>
	<id>[Your RIAForge project id]</id> 
	<username>[Your RIAForge username]</username>
	<password>[Your RIAForge password]</password>
	<name>[Your RIAForge project name]</name>
	<shortdescription>[A short explanation of the application]</shortdescription>
	<requirements>[External dependencies]</requirements> 
	<projectcategoryidfk>[The category ID] (See below)</projectcategoryidfk>
	<version>[The version of the application]</version> 
	<externalurl>[The url of your external url for this project]</externalurl>
	<externaldownload>[The url of your external download site]</externaldownload>
	<licenseidfk>[The license ID] (See below)</licenseidfk>
	<removecurrentfileflag>[Whether or not to remove the current file]</removecurrentfileflag>
	<newdownload>[The path for your zip file]</newdownload>
	<forums>[(TRUE|FALSE) whether or not to use RIAFORGE provided forums]</forums> 
	<externalforums>[The url of your external forums.]</externalforums>
	<blog>[(TRUE|FALSE) whether or not to use RIAFORGE provided blog]</blog> 
	<externalblog>[The url of your external blog]</externalblog>
	<externalblogrss>[The url of your external blog rss feed.]</externalblogrss>
	<bugtracker>[(TRUE|FALSE) whether or not to use RIAFORGE provided bugtracker]</bugtracker> 
	<externalbugtracker>[The url of your external bug tracker]</externalbugtracker>
	<wiki>[(TRUE|FALSE) whether or not to use RIAFORGE provided wiki]</wiki> 
	<externalwiki>[The url of your external wiki]</externalwiki>
	<svn>[(TRUE|FALSE) whether or not to use RIAFORGE provided svn]</svn> 
	<demourl>[The url of your external wiki]</demourl>
	<description>[A longer explanation of the application]</description>  
</config>
